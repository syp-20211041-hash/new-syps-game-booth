function renderLeaderboardUI(list, type) {
    const ul = document.getElementById('leaderboard-list'); 
    if (!ul) return;
    ul.innerHTML = '';

    // 📋 新規格 ④：排行榜清空後顯示全空狀態
    if (!list || list.length === 0) {
        ul.innerHTML = `
  • ${currentLang === 'zh' ? '目前排行榜全空，等待新挑戰者！' : 'Leaderboard is empty!'}
    • `;
        return;
    }

    list.forEach(function(player, index) {
        const li = document.createElement('li');
        
        // 📋 新規格 ③ & ⑤：過濾名字 (輸入 N/A、沒有、或留白，一律顯示 使用者(Number))
        let safeName = formatPlayerName(player.name, index + 1);
        
        // 📋 新規格 ①：只有當管理員權限開啟時才會顯示 ❌ 清除按鈕
        let deleteButton = '';
        if (isAdmin) {
            deleteButton = ``;
        }
        
        li.innerHTML = `No.${index + 1} - ${safeName} : ${player.score} 分${deleteButton}`;
        ul.appendChild(li);
    });
}

function formatPlayerName(inputName, num) {
    if (!inputName) return `使用者(${num})`;
    const check = inputName.trim().toUpperCase();
    if (check === 'N/A' || check === '沒有' || check === '') {
        return `使用者(${num})`;
    }
    return inputName;
}

function deleteLeaderboardItem(type, index) {
    if (!isAdmin) return;
    if (type === 'main') {
        let currentList = cloudState.leaderboard || [];
        currentList.splice(index, 1);
        dbRef.update({ leaderboard: currentList });
    } else {
        let targetRef = (type === 'shoot') ? 'scores_shoot' : 'scores_guess';
        let currentList = (type === 'shoot') ? shootHighScores : guessHighScores;
        currentList.splice(index, 1);
        database.ref(targetRef).set(currentList);
    }
}

function submitAdminCmd() {
    const input = document.getElementById('cmds-input');
    if (!input) return;
    const cmd = input.value.trim().toLowerCase();
    input.value = ''; 

    // 📋 新規格 ⑩：管理員輸入指令 begin Player 1 登入特定帳戶
    if (cmd.startsWith('begin player ')) {
        const pNum = cmd.replace('begin player ', '');
        if (['1', '2', '3'].includes(pNum)) {
            localStorage.setItem('game_currentUser', 'PLAYER' + pNum);
            dbRef.child(`players/${pNum}`).update({ loggedIn: true });
            alert(`[巨魔密令] 玩家 Player ${pNum} 帳戶已在後台被強制登入！`);
        }
    }
    // 📋 新規格 ⑨：限制手動登出，必須輸入指令 logout Player 1 才能登出
    else if (cmd.startsWith('logout player ')) {
        const pNum = cmd.replace('logout player ', '');
        if (['1', '2', '3'].includes(pNum)) {
            localStorage.removeItem('game_currentUser');
            dbRef.child(`players/${pNum}`).update({ loggedIn: false });
            alert(`[巨魔密令] 玩家 Player ${pNum} 已被強制登出！`);
        }
    }
    // 💥 巨魔加速外掛：set speed 999 
    else if (cmd === 'set speed 999') {
        dbRef.update({ speedMultiplier: 999 });
        alert("👹 巨魔外掛已發動！時間將在一秒內全部扣完！");
    }
    else if (cmd === 'set speed 1') {
        dbRef.update({ speedMultiplier: 1 });
        alert("時空流流速已恢復正常。");
    }
    else {
        alert("巨魔密令查無此功能，請重新確認指令！");
    }
}
